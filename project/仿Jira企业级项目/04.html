# JWT原理与auth-provider实现

src/auth-provider.ts

// 在真实环境中，如果使用firebase这种第三方auth服务的话，本文件不需要开发者开发

import { User } from 'screens/project-list/search-pannel';

const localStorageKey = '__auth_provider_token__';

export const getToken = () => window.localStorage.getItem(localStorageKey);

export const handleUserResponse = ({ user }: { user: User }) => {
	window.localStorage.setItem(localStorageKey, user.token || '');
	return user;
};

export const apiUrl = process.env.REACT_APP_API_URL;
export const login = (param: { username: string; password: string }) => {
	fetch(`${apiUrl}/login`, {
		method: 'POST',
		headers: {
			'Content-Type': 'application/json'
		},
		body: JSON.stringify(param)
	}).then(async (res) => {
		if (res.ok) {
      return handleUserResponse(await res.json())
		}
	});
};

export const register = (param: { username: string; password: string }) => {
	fetch(`${apiUrl}/register`, {
		method: 'POST',
		headers: {
			'Content-Type': 'application/json'
		},
		body: JSON.stringify(param)
	}).then(async (res) => {
		if (res.ok) {
      return handleUserResponse(await res.json())
		}
	});
};

export const logout = () => window.localStorage.removeItem(localStorageKey)

# 用useContext存储全局用户信息

src/context/auth-context.tsx

import React, { ReactNode, useState } from "react"
import { User } from "screens/project-list/search-pannel"
import * as auth from '../auth-provider'

interface AuthForm {
  username: string,
  password: string
}

const AuthContext = React.createContext<{
  user: User | null,
  login: (form: AuthForm) => Promise<void>,
  register: (form: AuthForm) => Promise<void>,
  logout: () => Promise<void>
} | undefined>(undefined)

AuthContext.displayName = 'AuthContext'

export const AuthProvider = ({children}: {children: ReactNode}) => {
  const [ user, setUser ] = useState<User | null>(null)

  // Argument of type 'User' is not assignable to parameter of type 'SetStateAction<null>'.
  const login = (form: AuthForm) => auth.login(form).then(user => setUser(user));
  const register = (form: AuthForm) => auth.register(form).then(setUser);
  const logout = () => auth.logout().then(() => setUser(null));

  // Type '{ user: User | null; login: (form: AuthForm) => Promise<void>; register: (form: AuthForm) => Promise<void>; logout: () => Promise<void>; }' is not assignable to type 'undefined'.
  return <AuthContext.Provider children={children} value={{user, login, register, logout}}></AuthContext.Provider>
}

export const useAuth = () => {
  const context = React.useContext(AuthContext)
  if (!context) throw new Error('useAuth必须在AuthProvider中使用')
  return context
}

# 用fetch抽象通用HTTP请求方法，增强通用性

下面的代码对 fetch 进行了封装，方便调用。

// src/utils/http.ts

import { toQuery } from "utils"
import * as auth from 'auth-provider'
import { useAuth } from "context/auth-context"

const apiUrl = process.env.REACT_APP_API_URL

interface Config extends RequestInit {
  data?: object,
  token?: string
}

// Property 'data' does not exist on type 'RequestInit'.
export const http = (endpoint: string, { data, token, headers, ...customConfig } : Config={}) => {
  const config = {
    method: 'GET',
    headers: {
      Authorization: token ? `Bearer ${token}` : '',
      'Content-Type': data ? 'application/json' : ''
    },
    ...customConfig
  }
  console.log('data', endpoint, data)
  if (config.method.toUpperCase() === 'GET') {
    endpoint += `?${toQuery(data)}`
  }else {
    config.body = JSON.stringify(data || {})
  }

  // axios 和 fetch 在返回状态不为 2xx 的时候表现不一样，fetch 不会抛出异常，axios 会抛出异常
  return window.fetch(`${apiUrl}/${endpoint}`, config)
  .then(async response => {
    if (response.status === 401) {
      await auth.logout()
      window.location.reload()
      return Promise.reject({ message: '请重新登录' })
    }
    const data = response.json()

    if (response.ok) return data
    
    return Promise.reject(data)
  })
}

export const useHttp = () => {
  const { user } = useAuth()
  // return ([endpoint, config]: [string, Config]) => http(endpoint, { ...config, token: user?.token })

  return (...[endpoint, config]: Parameters<typeof http>) => http(endpoint, { ...config, token: user?.token })
}

# 用useHttp管理JWT和登录状态，保持登录状态

下面的代码的作用是为了保持登录状态，解决用户刷新需要重新登陆的问题。














 
 
 
 
 
 
 
 
 


















 
 
 











// src/context/auth-context.tsx

import { useMount } from "hook/use-mount"
import React, { ReactNode, useState } from "react"
import { User } from "screens/project-list/search-pannel"
import { http } from "utils/http"
import * as auth from '../auth-provider'

interface AuthForm {
  username: string,
  password: string
}

const bootstrapUser = async () => {
   let user = null
   const token = auth.getToken()
   if (token) {
     const data = await http('me', {token})
     user = data.user
   }
   return user
}

const AuthContext = React.createContext<{
  user: User | null,
  login: (form: AuthForm) => Promise<void>,
  register: (form: AuthForm) => Promise<void>,
  logout: () => Promise<void>
} | undefined>(undefined)

AuthContext.displayName = 'AuthContext'

export const AuthProvider = ({children}: {children: ReactNode}) => {
  const [ user, setUser ] = useState<User | null>(null)

  // Argument of type 'User' is not assignable to parameter of type 'SetStateAction<null>'.
  const login = (form: AuthForm) => auth.login(form).then(user => setUser(user));
  const register = (form: AuthForm) => auth.register(form).then(setUser);
  const logout = () => auth.logout().then(() => setUser(null));

  useMount(() => {
    bootstrapUser().then(setUser)
  })

  // Type '{ user: User | null; login: (form: AuthForm) => Promise<void>; register: (form: AuthForm) => Promise<void>; logout: () => Promise<void>; }' is not assignable to type 'undefined'.
  return <AuthContext.Provider children={children} value={{user, login, register, logout}}></AuthContext.Provider>
}

export const useAuth = () => {
  const context = React.useContext(AuthContext)
  if (!context) throw new Error('useAuth必须在AuthProvider中使用')
  return context
}

# TS的联合类型、Partial和Omit介绍

# 联合类型

let a: string | number
a = 1
a = 'a'

如果我们再定义个变量 b 它和 a，我们可以把 a 复制份改写如下

let b: string | number
b = 1
b = 'a'

但如果我们还想定义 c、d、e... 如果都复制一份未免太麻烦，这个时候我们可以使用类型别名来修改

type T = string | number
let a:T = 1
let b:T = 1

是不是感觉和我们前面提到的 interface 很类似

type Person1 = {
  name: string,
  age: number
}

interface Person2 {
  name: string,
  age: number
}

const p1:Person1 = {
  name: 'f',
  age: 18
}

const p2:Person2 = {
  name: 'g',
  age: 17
}

上面可以看出使用两种方式都可以，在实际开发中，我们应该使用哪个呢？类型别名在很多情况下可以和 interface 互换使用。我们说一下，它们两个最主要的两处不同：

  • interface 不能定义联合、交叉类型
type T = string | number
let a: T = 'hello'
  • interface 没法实现 Utility type

我们之前封装的 http.ts 中的 useHttp方法中有这样一句代码：Parameters<typeof http>。细心的同学可能看到了 typeof 这个代码。注意：这里的 typeof 是 TS 中的，不是我们之前写 JS 使用到的 typeof 方法。JS 中的 typeof 是在 runtime 时运行的，TS 中的这个 typeof 是静态的，真正编译完后是不存在这个代码的。TS 中的 typeof 可以用来获取一个变量、对象、函数的类型。

interface Person1 {
  name: string,
  age: number
}

const person1: Person1 = {
  name: 'f',
  age: 18
}

// 对象
type Person2 = typeof person1 // type Person2 = Person1
const person2: Person2 = {
  name: 'g',
  age: 17
}

// 函数
function getValue(a: number):number {
  return a
}

type FN = typeof getValue // type FN = (a: number) => number

来看下 Parameters 的实现：

/**
 * Obtain the parameters of a function type in a tuple 获取元组中函数类型的参数
 */
type Parameters<T extends (...args: any) => any> = T extends (...args: infer P) => any ? P : never;

结合上面的知识点 Parameters<typeof http>，首先通过 typeof 获取 http 函数的类型，然后再通过 Parameters 获取这个对象的参数类型。

# Utility Types

Partial<Type> 可以将传入的类型属性变成可选的。

Omit<Type, Keys>

type Person = {
  name: string,
  age: number
}

// 将 Person 中的 name 和 age 都变成可选的，得到一个新 Type
const a: Partial<Person> = {}

// 从 Person 中拿出所有的属性，把 name 删除，得到一个新的 Type
const b: Omit<Person, 'age'> = { name: 'f'}
// 从 Person 拿出所有的属性，把 name 和 age 都删除，得到一个新的 Type
const c: Omit<Person, 'name' | 'age'> = {}

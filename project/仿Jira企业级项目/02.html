import { useEffect, useState } from "react"

export const SearchPannel = () => {
  const [params, setParams] = useState({
    name: '',
    personId: ''
  })

  const [users, setUsers] = useState([])
  const [list , setList] = useState([])

  useEffect(() => {
    fetch('').then(async => res => {
      if (res.ok) {
        setList(await res.json())
      }
    })
  }, [params])

  return <form action="">
    <div>
      <input val={params.name} onChange={e => setParams({
        ...params,
        name: e.target.value
      })}/>
      <select value={params.personId} onChange={e => setParams({
        ...params,
        personId: e.target.value
      })}>
        <option value={''}>负责人</option>
        {
          users.map(user => <option value={user.id}>{user.name}</option>)
        }
      </select>
    </div>
  </form>
}

# 【扩展学习】为什么 React 列表要加 key ?

React使用列表的 key 来进行对比，如果不指定，就默认为 index 下标。为什么 React 列表要添加 key 呢？

假设现在有这样一段代码：

const users = [{
  name: 'f'
}, {
  name: 'g'
}]

users.map((user, index) => <div key={index}>{ user.name }</div>)

它会渲染出下面这样的 DOM 树。

<div key="1">f</div>
<div key="2">g</div>

然后用户做了某个操作，users 被 unshift 另一个对象，变成：

const users = [
  {
    name: 'fg'
  },
  {
    name: 'f'
  },
  {
    name: 'g'
  }
]

DOM 树会变成下面这样，注意 key 的变化：

<div key="1">fg</div>
<div key="2">f</div>
<div key="3">g</div>

DOM 树的前后对比：

<div key="1">f</div> | <div key="1">fg</div>
<div key="2">g</div> | <div key="2">f</div>
                     | <div key="3">g</div>

在我们人类看来只不过是在开头加了一个 fg 而已。

但是 React 使用 key 值来识别变化，所以 React 认为的变化是：

f -> fg
g -> f
添加 g

这样非常消耗性能。

但是如果给 key 指定一个合适的值：

users.map((user, index) => <div key={user.name}>{ user.name }</div>)

React 认为的变化就变成：

                     | <div key="1">fg</div>
<div key="2">f</div> | <div key="2">f</div>
<div key="1">g</div> | <div key="3">g</div>

这样只需要做一个 unshift 操作，大大节省了性能。

# 学习自定义Hook - 用useDebounce减少工程搜索请求频率

  1. 自定义hook一定要使用use开头

  2. useMount的实现

export const useMount = (cb: () => void) => {
  useEffect(() => {
    cb();
  // 空数组表示只执行一次
  }, []);
};
  1. debounce的实现

作用：在特定时间内传入的 cb 函数只执行一次

const debounce = (cb, delay) => {
  let timer
  return (...params) => {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      cb(...params)
    }, delay)
  }
}

const log = debounce(()=>console.log('call'), 2000)

log();
log();
log();

// 运行结果，虽然上面调用了三次 log 函数，控制台只输出一次 'call'
  1. 自定义 useDebounce hook
import { useEffect, useState } from "react"

export const useDebounce = (value, delay) => {
  const [ debouncedValue, setDebouncedValue ] = useState(value)

  useEffect(() => {
    // 每次在 value 变化后设置一个定时器
    const timer = setTimeout(() => setDebouncedValue(value), delay)
    // 返回的这个函数在上一次的 useEffect 运行完后再运行，一般可以用于一些清理的工作
    return () => clearTimeout(timer)
  }, [value, delay])

  return debouncedValue
}

使用

const debouncedParam = useDebounce(param , 2000)
useEffect(() => {
  fetch(`${apiUrl}/projects?${toQuery(cleanObject(debouncedParam))}`)
  .then(async res => {
    if (res.ok) {
      setList(await res.json())
    }
  })
  // 当 debouncedParam 的值发生变化后再次执行 useEffect 中的代码
}, [debouncedParam])

什么时候需要使用自定义 hook 呢？从上面的 useDebounce 中我们总结点经验：

  • 这个函数内部需要使用到其他 hook。
  • 函数返回的值需要是响应式的。

# koa的基本使用

安装

npm i koa -S

新建app.js文件，并输入以下代码

const Koa = require('koa')
const app = new Koa()

app.use(async (ctx, next) => {
  ctx.body = 'hello world'
})

app.listen(5300)

运行

nodemon app.js

在浏览器访问http://localhost:5300/，浏览器页面会显示hello world文本，上面我们给前端返回的是一个文本，如果我们想要给前端返回JSON数据呢，只需要给保底赋值一个对象即可。

app.use(async (ctx, next) => {
  ctx.body = {name: 'f', age: 18}
})

下面我们来简单说下koa中间件，中间件实际就是个函数，我们在koa中可以通过use使用

const Koa = require('koa')
const app = new Koa()

app.use((ctx, next) => {
  console.log('执行了第一个中间件')
})

app.use((ctx, next) => {
  console.log('执行了第二个中间件')
})

app.listen(5300)

在浏览器输入http://localhost:5300/，此时终端会输出执行了第一个中间件，第二个中间件没有被执行，这是因为koa中会自动帮执行第一个中间件，后续的中间件需要开发者自己来调用。

ctx代表上下文

next代表下一个中间件，我们只需要在一个中间件中调用next()就可以执行下一个中间件了。

const Koa = require('koa')
const app = new Koa()

app.use((ctx, next) => {
  console.log('执行了第一个中间件')
  next()
})

app.use((ctx, next) => {
  console.log('执行了第二个中间件')
})

app.listen(5000)

输出结果

执行了第一个中间件
执行了第二个中间件

# 洋葱模型

下面代码运行会输出什么？

const Koa = require('koa')
const app = new Koa()

app.use((ctx, next) => {
  console.log(1)
  next()
  console.log(2)
})

app.use((ctx, next) => {
  console.log(3)
  next()
  console.log(4)
})

app.listen(5000)

答案

1
3
4
2


app.use(async (ctx, next) => {
  console.log(1)
  await next()
  console.log(2)
})

app.use(async (ctx, next) => {
  console.log(3)
  await next()
  console.log(4)
})

# 强制promise

next调用结果返回的是一个promise

app.use((ctx, next) => {
  console.log(next()) // Promise { 1 }， 如果第二个中间件没有返回值的话输出的是 Promise { undefined }, 返回的promise输出Promise { 'ok' }
})

app.use((ctx, next) => {
  // return 1 // 返回其他值也是可以的，比如返回一个promise
  return new Promise((resolve, reject) => {
    resolve('ok')
  })
})

await可以理解为一个求值，他会把promise成功的值返回给某个变量

await会阻塞线程

如果在一个函数的前面加上async，这个函数任何的返回值都会被包装成promise

async function fn() {
  return 1
}
console.log(fn()) // Promise {<fulfilled>: 1}

# 为什么一定要保证洋葱模型

所有的中间件都要加上async和await，如果不加的话很难保证所有的中间件都是按照洋葱模型执行的。

app.use((ctx, next) => {
  console.log(1)
  next()
  console.log(2)
})

app.use(async (ctx, next) => {
  console.log(3)
  await next()
  console.log(4)
})

输出结果

1
3
2
4

可以看出输出结果和我们上面分析的不一样了。

洋葱模型是以next为分解线的

下面这样拿不到请求的结果，不加async和await无法保证后面的代码已经执行了，所以拿不到结果

app.use((ctx, next) => {
  next()
  const r = ctx.r
  console.log(r) // undefined
})

app.use(async (ctx, next) => {
  const axios = require('axios')
  const res = await axios.get('https://baidu.com')
  ctx.r = res
  await next()
})

下面这样能拿到请求的结果。

app.use(async (ctx, next) => {
  await next()
  const r = ctx.r
  console.log(r)
})

app.use(async (ctx, next) => {
  const axios = require('axios')
  const res = await axios.get('https://baidu.com')
  ctx.r = res
  await next()
})

const requireDirectory = require('require-directory')

const app = new Koa()

function whenLoadModule(obj) {

    app.use(obj.routes())
  }
}

# 初始化管理器与Process.cwd

const Router = require('koa-router')
const requireDirectory = require('require-directory')

class InitManager {
  static initCore(app) {
    InitManager.app = app
    InitManager.initLoadRouters()
  }

  static initLoadRouters() {
    requireDirectory(module, `${process.cwd()}/app/api`, { visit: whenLoadModule })
    
    function whenLoadModule(obj) {
      if (obj instanceof Router) {
        InitManager.app.use(obj.routes())
      }
    }
  }
}

module.exports = InitManager

# 全局异常处理中间件编写

有空理解下AOP（面向切面编程）

全局异常处理的实现：

const { env } = require('../config/config')
const { HttpException } = require('../core/http-exception')
const catchError = async (ctx, next) => {
  try {
    await next()
  }catch(err) {
    const isHttpException = err instanceof HttpException
    const isDev = env == 'dev'
    // 开发环境抛出错误，方便后端调试
    if (isDev && !isHttpException) {
      throw err
    }

    if (isHttpException) {
      ctx.body = {
        msg: err.msg,
        errorCode: err.errorCode,
        request: `${ctx.method} ${ctx.path}`
      }
      ctx.status = err.code
    }
  }
}

module.exports = catchError

使用

// app.js
const catchError = require('./middlewares/exception')
app.use(catchError)

通过ctx.method可以获取到请求的方法，通过ctx.status可以设置前端的http状态码

# Sequelize初始化配置与注意事项

安装

cnpm i sequelize mysql2 -S

配置sequelize

// db.js
const Sequelize = require('sequelize');

const { dbName, host, port, user, password } = require('../config/config').database;

const sequelize = new Sequelize(dbName, user, password, {
	dialect: 'mysql', // 用于指定数据库的类型
	host,
	port,
	logging: true, // 打印原始sql语句
	timezone: '+08:00', // 设置时区
	define: {
    timestamps: true, // 设置表中是否显示createdAt、updatedAt字段，设置为false时不显示这两个字段
    paranoid: true, // 显示deletedAt字段
    createdAt: 'created_at', // 更改自动生成的字段名
    updatedAt: 'updated_at',
    underscored: true // 将字段名由驼峰改为下划线连接的方式
  }
});

// 如果不加这句sequelize不会自动把模型创建到数据库的
sequelize.sync({
  force: false // 尽量不要设置为true，会自动删除表从新创建
})
 
module.exports = { sequelize };

sequelize.sync({ force: false })尽量不要设置为true，会自动删除表从新创建

给require引入的模块起个别名:

// 这样就可以直接使用db调用sequelize导出的方法或者属性了
const { sequelize: db } = require('../../core/db');

# User模型与用户唯一标识设计探讨

当我们在设计表的主键的时候要注意：1.主键不能为空 2.主键不能重复 3.要设计成数字的，查询效率比较高。 4.不要使用随机字符串，典型的随机字符串一般使用GUID，因为数据量大的时候查询比较慢。因为数据量大的时候查询比较慢。

// models/user.js

const { Sequelize, Model } = require('sequelize');
const { sequelize } = require('../../core/db');

class User extends Model {
  static async verifyEmailPassword(email, plainPassword) {}
}

User.init(
	{
		id: {
			type: Sequelize.INTEGER,
			primaryKey: true,
			autoIncrement: true
		},
		nickname: Sequelize.STRING, // 没有其他限制的话也可以这样写

      type: Sequelize.STRING(128),
      unique: true

      type: Sequelize.STRING,
      // 这样的写法实际是观察者模式的一种使用
      // es6中可以使用Reflect实现观察者模式

        const salt = bcryptjs.genSaltSync(10)
        const pwd = bcryptjs.hashSync(val, salt)
        this.setDataValue('password', pwd)
      }
    },

      解决办法：
      1.手动删除原有表，再次执行代码
      2.配置sequelize.sync，如下配置可以强制删除原有表，新建个新表
      sequelize.sync({
        force: true
      })
      以上两种方法最好不要使用，因为都是删除已有表，有可能造成数据丢失
      可以在表上右键单击选择设计表，进行相关字段的增删

    // test2: Sequelize.STRING,

			type: Sequelize.STRING(64),
			unique: true
		}
		// 一个用户同一个小程序的openid是唯一的，两个小程序的openid是不一样的
		// unionID是唯一的
	},

    sequelize,
    tableName: 'user' // 命名表名
	}

# 验证JWT令牌

const basicAuth = require('basic-auth')
const jwt = require('jsonwebtoken')
const config = require('../config/config')
const { Forbbiden } = require('../core/http-exception')

class Auth {
    constructor(level) {
        this.level = level || 1
        Auth.USER = 8
        Auth.ADMIN = 16
        Auth.SUPER_ADMIN = 32
    }

    get m() {
        return async (ctx, next) => {
          console.log('执行了')
            // ctx.req后去的是node.js原生的request对象
            // ctx.request获取的是koa对node.js原生request封装的一个对象
            const userToken = basicAuth(ctx.req)
            ctx.body = userToken
            let errMsg = 'token不合法'

            if (!userToken || !userToken.name) {
                throw new Forbbiden(errMsg)
            }

            try {
              // 校验令牌
                var decode = jwt.verify(userToken.name, 
                    config.security.secretKey)
            } catch (error) {
                if (error.name == 'TokenExpiredError'){
                    errMsg = 'token已过期'
                }
                throw new Forbbiden(errMsg)
            }

            if(decode.scope < this.level){
                errMsg = '权限不足'
                throw new global.errs.Forbbiden(errMsg)
            }

            // uid,scope 将自定义属性保存到ctx.auth属性上
            ctx.auth = {
                uid:decode.uid,
                scope:decode.scope
            }

            await next()
        }
    }

    static verifyToken(token){
        try{
            jwt.verify(token, 
                config.security.secretKey)
            return true
        }
        catch (error){
            return false
        }
    }
}

module.exports = {
    Auth
}

# 数据库事务的基本概念

一个设计比较严谨的数据库一般会有四个最基本的特性：ACID表示原子性 一致性 隔离性 持久性

如点赞业务， 一次操作需要操作两张数据表

  1. favor添加是否喜欢记录
  2. 操作其他表里的fav_nums

必须确保两个都执行，不能出现一个执行一个不执行的情况，可以使用数据库事务，数据库事务可以保证数据一致性

# 点赞业务逻辑实现

// app/models/favor.js
const { sequelize } = require('../../core/db');
const { Sequelize, Model, Op } = require('sequelize');
const { Art } = require('./art');
const { LikeError, DislikeError } = require('../../core/http-exception');

// favor是一个业务表
class Favor extends Model {
  /* 
  一次操作需要操作两张数据表
  1.favor添加是否喜欢记录
  2.操作其他表里的fav_nums
  必须确保两个都执行，不能出现一个执行一个不执行的情况，可以使用数据库事务，数据库事务可以保证数据一致性
  */
	static async like(art_id, type, uid) {
		const favor = await Favor.findOne({
			where: {
				art_id,
				type,
				uid
			}
		});
    // 如果有值表明用户已经点过赞了
		if (favor) {
			throw new LikeError();
		}
    // 执行事务
		return sequelize.transaction(async (t) => {
      // 往数据表中插入一条数据
			await Favor.create(
				{
					art_id,
					type,
					uid
				},
				{
					transaction: t
				}
			);
      // 查看用户操作的哪种媒体哪个id
			const art = await Art.getData(art_id, type);

      // 对数字进行加一
			await art.increment('fav_nums', {
				by: 1, // 增加多少
				transaction: t
			});
		});
	}
	static async disLike(art_id, type, uid) {
		const favor = await Favor.findOne({
			where: {
				art_id,
				type,
				uid
			}
		});
		if (!favor) {
			throw new DislikeError();
		}
		// Favor 表 favor 记录
		return sequelize.transaction(async (t) => {
      // 删除一条记录，对查询出来的favor进行destroy
			await favor.destroy({
				force: true, // false软删除 true物理删除
				transaction: t
			});
			const art = await Art.getData(art_id, type, false);
      // 减一
			await art.decrement('fav_nums', {
				by: 1,
				transaction: t
			});
		});
	}
}

Favor.init(
	{
		uid: Sequelize.INTEGER, // 表示用户id
		art_id: Sequelize.INTEGER, // 用户对哪个媒体进行了点赞
		type: Sequelize.INTEGER // 用户对哪种类型的媒体进行了点赞
	},
	{
		sequelize,
		tableName: 'favor'
	}
);

module.exports = {
	Favor
};

# Sequelize Scope的应用

要过滤前端不使用的字段有三种方法，一种是直接从对象中删除；另一种是scope；最后一中是json序列化

这里使用scope排除一些不用的字段，当然scope也可以放到每个具体的模型中，但是那样太麻烦

// core/db.js
const sequelize = new Sequelize(dbName, user, password, {
  // ...
	define: {
    // ...
		scopes: {
			bh: {// bh名字可以任意选取
				attributes: {// 返回给前端的结果中排除下面三个字段
					exclude: [ 'updated_at', 'deleted_at', 'created_at' ]
				}
			}
		}
	}
});

使用上面定义的scope，排除不需要的返回结果中的字段
